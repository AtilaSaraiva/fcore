135  ! Initialization

138  else
139  errmgr => deferr
140  end if
141 
142  ! Process
143  index = this%index_of_key(key)
144  if (index <= 0) then
145  ! ERROR
146  call errmgr%report_error("dict_set", &
147  "The specified key could not be found in the dictionary.", &
148  fcore_nonexistent_key_error)
149  return
150  end if
151 
152  ! Replace the item
153  ptr => this%m_list%get(index)
154  select type (ptr)
155  class is (key_value_pair)
156  ! Delete the old object
157  if (associated(ptr%value)) deallocate(ptr%value)
158 
159  ! Create a copy of item, and store that copy
160  allocate(cpy, source = item, stat = flag)
161  if (flag /= 0) then
162  call errmgr%report_error("dict_set", &
163  "Insufficient memory available.", &
164  fcore_out_of_memory_error)
165  return
166  end if
167  ptr%value => cpy
168  call this%m_list%set(index, ptr, errmgr)
169  if (errmgr%has_error_occurred()) return
170  end select
171  end subroutine
